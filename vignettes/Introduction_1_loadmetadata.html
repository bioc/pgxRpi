Load library

library(pgxRpi)

pgxLoader function in this package

This function loads various data from Progenetix database.

The parameters of this function:

  • type A string specifying output data type. Available options are “biosample”, “individual”, “variant” or “frequency”.
  • output A string specifying output file format. When the parameter type is “variant”, available options are NULL, “pgxseg” ,“pgxmatrix”, “coverage” or “seg” ; When the parameter type is “frequency”, available options are “pgxfreq” or “pgxmatrix” .
  • filters Identifiers for cancer type, literature, cohorts, and age such as c(“NCIT:C7376”, “pgx:icdom-98353”, “PMID:22824167”, “pgx:cohort-TCGAcancers”, “age:>=P50Y”). Multiple filters are combined using an OR logic, except for filters associated with age, which use an AND logic when combined with other filters.
  • codematches A logical value determining whether to exclude samples from child concepts of specified filters that belong to cancer type/tissue encoding system (NCIt, icdom/t, Uberon). If TRUE, retrieved samples only keep samples exactly encoded by specified filters. Do not use this parameter when filters include cancer-irrelevant filters such as PMID and cohort identifiers. Default is FALSE.
  • limit Integer to specify the number of returned samples/individuals/coverage profiles for each filter. Default is 0 (return all).
  • skip Integer to specify the number of skipped samples/individuals/coverage profiles for each filter. E.g. if skip = 2, limit=500, the first 2*500 =1000 profiles are skipped and the next 500 profiles are returned. Default is NULL (no skip).
  • biosample_id Identifiers used in Progenetix database for identifying biosamples.
  • individual_id Identifiers used in Progenetix database for identifying individuals.
  • save_file A logical value determining whether to save the variant data as file instead of direct return. Only used when the parameter type is “variant” and output is “pgxseg” or “seg”. Default is FALSE.
  • filename A string specifying the path and name of the file to be saved. Only used if the parameter save_file is TRUE. Default is “variants.seg/pgxseg” in current work directory.

The data contains many columns representing different aspects of sample information. By default, it returns all related samples (limit=0). You can access a subset of them via the parameter limit and skip. For example, if you want to access the first 1000 samples , you can set limit = 1000, skip = 0.

biosamples_2 <- pgxLoader(type="biosample", filters = "NCIT:C3512",skip=0, limit = 1000)
# Dimension: Number of samples * features
print(dim(biosamples))
#> [1] 4642   53
print(dim(biosamples_2))
#> [1] 1000   53

The number of samples in specific group can be queried by pgxCount function.

pgxCount(filters = "NCIT:C3512")
#>      filters               label total_count exact_match_count
#> 1 NCIT:C3512 Lung Adenocarcinoma        4642              4506

The NCIt code of retrieved samples doesn’t only contain specified filters but contains child terms.

Setting codematches as TRUE allows this function to only return biosamples with exact match to the filter.

#> [1] "NCIT:C3512"

Survival data use case

Cancer-specific query

To get metadata from individuals with specific cancer, you can use corresponding disease code as filters. Here we use NCIt code of Lung Adenocarcinoma (C3512) as an example and analyse survival difference between older patients ( >= 60 years old) and younger patients ( < 60 years old).

# load required library
library(lubridate)
#>

#> 
#>     date, intersect, setdiff, union
library(survival)
library(survminer)
#> Loading required package: ggplot2
#> Loading required package: ggpubr
#> 
#> Attaching package: 'survminer'
#> The following object is masked from 'package:survival':
#> 
#>     myeloma

# query
lungadind_young <- pgxLoader(type='individual',filters=c('NCIT:C3512','age:<P60Y'),codematches = F)
lungadind_old <- pgxLoader(type='individual',filters=c('NCIT:C3512','age:>=P60Y'),codematches = F)

Define function for preprocessing data

cleanData <- function(data){
  # exclude individuals without available survival information
  data <- data[data$index_disease_followup_time != "None" & 
                       !data$index_disease_followup_state_label %in% c("","no followup status"),]
  # transform followup state 
  aliveidx <- which(data$index_disease_followup_state_label == "alive (follow-up status)")
  data$index_disease_followup_state_label[aliveidx] <- 0 
  deadidx <- which(data$index_disease_followup_state_label == "dead (follow-up status)")
  data$index_disease_followup_state_label[deadidx] <- 1
  return(data)
}

Preprocess data and construct survival data frame

lungadind_young_clean <- cleanData(lungadind_young)
lungadind_old_clean <- cleanData(lungadind_old)
lungadind_clean <- rbind(lungadind_young_clean,lungadind_old_clean)

group_id <- rep(c('younger','older'),times=c(dim(lungadind_young_clean)[1],dim(lungadind_old_clean)[1]))
followup_state <- as.numeric(lungadind_clean$index_disease_followup_state_label)
followup_time <- time_length(lungadind_clean$index_disease_followup_time,unit = 'days')
survival_df <- data.frame(group_id, followup_state, followup_state)

Fit the survival data and visualise

sfit <- survfit(Surv(followup_time, followup_state)~group_id, data=survival_df)
ggsurvplot(sfit,data=survival_df,pval = T)

Cohort-specific query

To get metadata from individuals with specific cohort, you need to use corresponding filters. Here we use TCGA-BRCA cohort from TCGA as an example and do the same survival analysis as above.

# query
tcgabrcaind_young <- pgxLoader(type='individual',filters=c('pgx:TCGA.BRCA','age:<P60Y'))
tcgabrcaind_old <- pgxLoader(type='individual',filters=c('pgx:TCGA.BRCA','age:>=P60Y'))
# preprocess
tcgabrcaind_young_clean <- cleanData(tcgabrcaind_young)
tcgabrcaind_old_clean <- cleanData(tcgabrcaind_old)
# survival data construction
tcgabrcaind_clean <- rbind(tcgabrcaind_young_clean,tcgabrcaind_old_clean)
group_id <- rep(c('younger','older'),times=c(dim(tcgabrcaind_young_clean)[1],dim(tcgabrcaind_old_clean)[1]))
followup_state <- as.numeric(tcgabrcaind_clean$index_disease_followup_state_label)
followup_time <- time_length(tcgabrcaind_clean$index_disease_followup_time,unit = 'days')
survival_df <- data.frame(group_id, followup_state, followup_state)
# fit
sfit <- survfit(Surv(followup_time, followup_state)~group_id, data=survival_df)
# visualise
ggsurvplot(sfit,data=survival_df,pval = T)
